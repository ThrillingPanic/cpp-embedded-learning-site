Bit Manipulation Patterns

Difficulty Level: Critical (Intermediate)

Bit manipulation is the language of hardware. Every hardware register is just a collection of bits. Every protocol encodes data bitwise. Every optimization starts with understanding bits. If you only learn one "pattern" topic in embedded, make it this one.

Why Bit Manipulation Is Critical

Embedded Reality: You control devices through bit patterns. Want LED on? Set bit 5 of GPIO register. Want interrupt enabled? Set bit 12 of control register. Want to extract temperature from a multi-byte sensor reading? Shift and mask bits. Every embedded system you program will depend on bitwise operations.

⚠️ Performance Fact: Bitwise operations are single CPU cycles. Branch instructions can cause pipeline stalls. For embedded systems with limited power, a single bitwise AND is preferred over an if statement every time.

1. Bitwise Operators Review

AND (&) - Select Specific Bits

AND produces 1 only if BOTH bits are 1. Use it to "mask out" (zero) unwanted bits:

uint8_t status = 0b11010110;
uint8_t mask = 0b00001111;
uint8_t lower_nibble = status & mask;

if (status & (1 << 3)) {
}

uint16_t sensor = 0x1A5C;
uint8_t low_byte = sensor & 0xFF;

OR (|) - Set Specific Bits

OR produces 1 if EITHER bit is 1. Use it to "turn on" specific bits:

uint8_t control = 0b00000000;
control |= (1 << 0);
control |= (1 << 3);

uint32_t gpio_config = 0;
gpio_config |= 0x0001;
gpio_config |= 0x0010;

XOR (^) - Toggle/Compare Bits

XOR produces 1 if bits DIFFER. Use it to toggle (flip) specific bits:

uint8_t led_state = 0;
led_state ^= (1 << 5);  // Toggle bit 5: becomes 0b00100000
led_state ^= (1 << 5);  // Toggle again: becomes 0b00000000

// Flash an LED: toggle every loop iteration
while(1) {
    led_state ^= (1 << LED_PIN);
    delay_ms(500);
}

NOT (~) - Invert All Bits

NOT flips every bit (0→1, 1→0). Use carefully—understand widths!

uint8_t mask = 0x0F;
uint8_t inverted = ~mask;

uint8_t config = 0xFF;
config &= ~(1 << 3);

Left Shift (<<) - Multiply by Powers of 2

Shift bits left N positions. Each shift multiplies by 2:

uint8_t value = 3;
uint8_t shifted = value << 2;

uint32_t bit_5_mask = 1 << 5;
uint32_t bit_15_mask = 1 << 15;

uint32_t packed = (uint8_t)0x12 << 24 |
                  (uint8_t)0x34 << 16 |
                  (uint8_t)0x56 << 8 |
                  (uint8_t)0x78;

Right Shift (>>) - Divide by Powers of 2

Shift bits right N positions. For unsigned, fills with 0. For signed, depends on implementation (usually sign-extends):

uint16_t value = 1024;
uint16_t divided = value >> 2;

uint32_t packed = 0x12345678;
uint8_t byte0 = (packed >> 24) & 0xFF;
uint8_t byte1 = (packed >> 16) & 0xFF;
uint8_t byte2 = (packed >> 8) & 0xFF;
uint8_t byte3 = packed & 0xFF;

int8_t negative = -4;
int8_t shifted = negative >> 1;

2. Common Bit Manipulation Patterns

Pattern 1: Check if Bit N is Set

#define BIT_IS_SET(value, bit) (((value) >> (bit)) & 1)

uint8_t status = 0b10110101;
if (BIT_IS_SET(status, 2)) {
}

if ((status & (1 << 2)) != 0) {
}

volatile uint32_t *interrupt_status = (volatile uint32_t*)0x40000008;
if (*interrupt_status & (1 << 5)) {
}

Pattern 2: Set Bit N

#define BIT_SET(value, bit) ((value) |= (1 << (bit)))

uint8_t flags = 0;
BIT_SET(flags, 3);  // Set bit 3: flags = 0b00001000

// Or without macro:
flags |= (1 << 3);

// For hardware: enable GPIO output on pin 5
volatile uint32_t *gpio_config = (volatile uint32_t*)0x40020000;
*gpio_config |= (1 << 5);  // Set output enable for pin 5

Pattern 3: Clear Bit N

#define BIT_CLEAR(value, bit) ((value) &= ~(1 << (bit)))

uint8_t flags = 0xFF;  // All bits set
BIT_CLEAR(flags, 3);   // Clear bit 3: flags = 0b11110111

// Or without macro:
flags &= ~(1 << 3);

// Practical: disable interrupt source 2
volatile uint32_t *interrupt_enable = (volatile uint32_t*)0x40000004;
*interrupt_enable &= ~(1 << 2);  // Disable interrupt 2

Pattern 4: Toggle Bit N

#define BIT_TOGGLE(value, bit) ((value) ^= (1 << (bit)))

uint8_t led = 0;
BIT_TOGGLE(led, 5);  // Set bit 5
BIT_TOGGLE(led, 5);  // Clear bit 5 again

// Blinking LED in main loop
while(1) {
    BIT_TOGGLE(gpio_output, LED_PIN);
    delay_ms(200);
}

Pattern 5: Extract N-Bit Field

// Extract bits 4-7 (4-bit field at position 4)
uint8_t value = 0b11010110;
uint8_t field = (value >> 4) & 0x0F;  // Shift right 4, mask 4 bits
// Result: 0b00001101 = 13

// Macro for clarity
#define EXTRACT_BITS(value, start, count) (((value) >> (start)) & ((1 << (count)) - 1))

uint8_t field2 = EXTRACT_BITS(value, 4, 4);  // Extract 4 bits starting at bit 4

// Real hardware example: ADC priority field (bits 5-7) of control register
uint8_t control_reg = 0b10110101;
uint8_t priority = EXTRACT_BITS(control_reg, 5, 3);  // Get priority (bits 5,6,7)

Pattern 6: Set N-Bit Field

uint8_t value = 0b00101101;
value &= 0x0F;
value |= (0x0A << 4);

value = (value & 0x0F) | (0x0A << 4);

#define SET_BITS(value, start, count, new_val) \
    ((value) = ((value) & ~(((1 << (count)) - 1) << (start))) | ((new_val) << (start)))

SET_BITS(control_reg, 5, 3, 3);

Pattern 7: Check if Power of 2

bool is_power_of_2(uint32_t n) {
    return (n > 0) && ((n & (n - 1)) == 0);
}

Pattern 8: Count Set Bits (Bit Population)

int count_set_bits(uint8_t value) {
    int count = 0;
    while (value) {
        count += value & 1;
        value >>= 1;
    }
    return count;
}

int fast_count_bits(uint8_t value) {
    int count = 0;
    while (value) {
        value &= (value - 1);
        count++;
    }
    return count;
}

int bit_count = __builtin_popcount(value);

uint32_t enabled_mask = 0x00001F03;
int active_interrupts = __builtin_popcount(enabled_mask);  // Returns 6

3. Real Embedded Patterns

Pattern: GPIO Pin Control

// STM32 GPIO example
#define GPIO_OUTPUT_ENABLE(port, pin) ((port) |= (1 << (pin)))
#define GPIO_OUTPUT_DISABLE(port, pin) ((port) &= ~(1 << (pin)))
#define GPIO_PIN_ON(port, pin) ((port) |= (1 << (pin)))
#define GPIO_PIN_OFF(port, pin) ((port) &= ~(1 << (pin)))
#define GPIO_PIN_TOGGLE(port, pin) ((port) ^= (1 << (pin)))

// Usage
uint32_t port_a = 0;
GPIO_PIN_ON(port_a, 5);      // LED on
GPIO_PIN_TOGGLE(port_a, 5);  // LED off
GPIO_PIN_TOGGLE(port_a, 5);  // LED on again

Pattern: Status Register Parsing

// Device status register (fictional)
// Bit 0: Ready
// Bits 1-3: Error code
// Bit 4: Busy flag
// Bits 5-7: Reserved

uint8_t status = read_device_status();

bool ready = status & 0x01;  // Check bit 0
uint8_t error_code = (status >> 1) & 0x07;  // Extract bits 1-3
bool busy = (status >> 4) & 0x01;  // Check bit 4

if (ready && !busy) {
    // Device is ready to use
    if (error_code != 0) {
        // But there's an error code
        printf("Error %d occurred\n", error_code);
    }
}

Pattern: Data Serialization

struct SensorData {
    uint16_t temp;
    uint8_t humidity;
    bool alert;
};

uint8_t encode_sensor_data(SensorData *data) {
    uint16_t packed = 0;
    
    packed |= (data->temp << 0);
    
    packed |= (data->humidity << 7);
    
    packed |= ((data->alert ? 1 : 0) << 14);
    
    return packed;
}

SensorData decode_sensor_data(uint16_t packed) {
    SensorData data;
    data.temp = (packed >> 0) & 0x7F;
    data.humidity = (packed >> 7) & 0x7F;
    data.alert = (packed >> 14) & 0x01;
    return data;
}

Performance Optimization

Bitwise operations are incredibly fast—usually single CPU cycles. Compare:

if (status == 1) {
}

if (status & 0x01) {
}
✅ Bit Mastery Levels:
  1. Beginner: Can check/set/clear individual bits
  2. Intermediate: Can extract and set multi-bit fields
  3. Expert: Optimizes data layouts using bit packing, understands byte ordering and bit operations at instruction level

Best Practices

  • Use macros for clarity: `BIT_SET(val, 3)` is clearer than `val |= 8`
  • Comment bit layouts: When dealing with packed data, document the bit layout
  • Consider endianness: Bit shifting behaves differently on big-endian vs little-endian systems
  • Use fixed-width types: Always use uint8_t, uint16_t, etc. Never assume int size
  • Beware of signed shifts: Right-shifting negative numbers can have surprising results
  • Test on hardware: Bit behaviors vary subtly across processors

Summary

Bit manipulation is the foundation of embedded programming. Master these operators and patterns:

  • & selects bits (AND)
  • | enables bits (OR)
  • ^ toggles bits (XOR)
  • ~ inverts bits (NOT)
  • << shifts left (multiply by 2^n)
  • >> shifts right (divide by 2^n)

Use them to control hardware, parse data, optimize memory, and communicate with devices. Every embedded system you'll ever program depends on these operations.
