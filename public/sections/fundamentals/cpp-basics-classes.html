Structs & Classes: Organizing Data and Behavior Together

The fundamental problem we're solving: In C, when you have related data, you scatter it across separate variables. A motor controller needs speed, direction, acceleration, and current readings—four separate variables. This is error-prone: you might forget to update one, pass the wrong variable, or lose track of which variables belong together.

The solution: Structs and Classes bundle related data (variables) and functions (methods) that operate on that data into a single unit. Instead of motor_speed, motor_direction, motor_acceleration, motor_current—four separate variables—you have ONE Motor struct containing all four, plus functions that operate on them.

Struct vs Class

In C++, the main difference is access control:

  • struct: Members are PUBLIC by default (anyone can access them)
  • class: Members are PRIVATE by default (only the class can access them)

For embedded systems where simplicity and transparency matter, structs are often sufficient. Use classes when you need data hiding and abstraction.

Real embedded examples where structs/classes organize complexity:

  • Peripheral driver: A UART driver struct contains the port address, baud rate, interrupt flags, and functions like send_byte() and read_byte()
  • Sensor abstraction: A Temperature Sensor struct contains pin, calibration data, and functions like read() and calibrate()
  • State machine: A Motor Controller struct contains state, speed, direction, and functions like set_speed(), set_direction(), update()
  • Configuration data: Device settings packed into struct for easy serialization to EEPROM

struct MotorController {
    uint8_t pin_pwm;       // PWM pin number
    uint8_t pin_direction; // Direction pin
    uint8_t speed;         // Current speed (0-255)
    uint8_t direction;     // 0 = forward, 1 = backward
    uint16_t max_rpm;      // Motor specification

// Create and use motor instance
MotorController motor1;
motor1.pin_pwm = 9;
motor1.pin_direction = 8;
motor1.speed = 100;
motor1.direction = 0;
motor1.max_rpm = 3000;

Memory Layout of Structs

struct Motor {
    uint8_t speed;      // 1 byte at offset 0
    uint8_t direction;  // 1 byte at offset 1
    uint16_t rpm;       // 2 bytes at offset 2-3
};

// Total size: 4 bytes (1 + 1 + 2)
// This is important! A struct's size is sum of its members' sizes
printf("Motor struct size: %d bytes\n", sizeof(Motor));  // 4
Embedded Reality: EEPROM Storage

Knowing struct size is critical for storing data in EEPROM (persistent storage). You need to know exactly how many bytes to read/write:

struct DeviceConfig {
    uint8_t device_id;
    uint16_t calibration_offset;
    uint32_t power_on_count;
};

// Store config to EEPROM (64 bytes offset)
EEPROM_write(64, (uint8_t*)&config, sizeof(DeviceConfig));
config_size = sizeof(DeviceConfig);  // 7 bytes

Adding Functions to Structs (Methods)

Functions That Work on Structs

In C, you pass the struct pointer to functions:

void motor_set_speed(MotorController *motor, uint8_t new_speed) {
    motor->speed = new_speed;
    analogWrite(motor->pin_pwm, new_speed);
}

void motor_set_direction(MotorController *motor, uint8_t dir) {
    motor->direction = dir;
    digitalWrite(motor->pin_direction, dir);
}

// Usage
MotorController motor = {9, 8, 0, 0, 3000};
motor_set_speed(&motor, 150);
motor_set_direction(&motor, 0);

C++ Class Methods (Cleaner Syntax)

In C++, methods are part of the class:

class Motor {
public:
    uint8_t speed;
    uint8_t pin_pwm;
    
    void set_speed(uint8_t new_speed) {
        speed = new_speed;
        analogWrite(pin_pwm, speed);
    }
};

// Usage (no need to pass &motor)
Motor motor;
motor.speed = 0;
motor.pin_pwm = 9;
motor.set_speed(150);  // 'this' pointer implicit

Real Example: Sensor Class

class TemperatureSensor {
public:
    uint8_t pin;
    int16_t calibration_offset;
    
    // Constructor: runs when object is created
    TemperatureSensor(uint8_t sensor_pin) {
        pin = sensor_pin;
        calibration_offset = 0;
        pinMode(pin, INPUT);
    }
    
    // Method: read current temperature
    int16_t read() {
        int raw = analogRead(pin);
        return (raw * 500) / 1024 + calibration_offset;
    }
    
    // Method: calibrate
    void set_calibration(int16_t offset) {
        calibration_offset = offset;
    }
};

// Usage
TemperatureSensor temp_sensor(A0);
int16_t current_temp = temp_sensor.read();
temp_sensor.set_calibration(-5);

Encapsulation: Hiding Implementation Details

The power of classes: You can make some data private (hidden from users) and provide public methods to access it safely.

Private vs Public Members

class UART {
private:
    // Private: only UART methods can access these
    uint32_t baud_rate;
    uint8_t buffer[256];
    uint8_t buffer_index;
    
public:
    // Public: anyone can call these
    void init(uint32_t baud) {
        baud_rate = baud;
        buffer_index = 0;
        // Configure hardware
    }
    
    void send(uint8_t data) {
        buffer[buffer_index++] = data;
    }
    
    uint8_t receive() {
        // Return data from buffer
        return buffer[0];
    }
};

// User code:
UART uart;
uart.init(9600);      // OK—public method
uart.send(0x42);      // OK—public method
// uart.baud_rate = 115200;  // ERROR! Private—not allowed

Real Embedded Pattern: GPIO Abstraction

class DigitalPin {
private:
    uint8_t pin;
    uint8_t mode;  // INPUT or OUTPUT
    
public:
    DigitalPin(uint8_t pin_number) : pin(pin_number) {
        mode = INPUT;  // Default
    }
    
    void set_output() {
        mode = OUTPUT;
        pinMode(pin, OUTPUT);
    }
    
    void set_input() {
        mode = INPUT;
        pinMode(pin, INPUT);
    }
    
    void write(uint8_t value) {
        digitalWrite(pin, value);
    }
    
    uint8_t read() {
        return digitalRead(pin);
    }
};

// Usage: simple, clean interface
DigitalPin led(13);
led.set_output();
led.write(1);  // Turn on

Constructors and Initialization

A constructor is a special function that runs automatically when an object is created:

Constructor Basics

class Motor {
public:
    uint8_t pin_pwm;
    uint8_t speed;
    
    // Constructor: runs automatically on creation
    Motor(uint8_t pwm_pin) {
        pin_pwm = pwm_pin;
        speed = 0;
        pinMode(pin_pwm, OUTPUT);
        digitalWrite(pin_pwm, LOW);  // Start off
    }
};

// Create motor—constructor runs automatically
Motor motor1(9);    // Constructor called with 9
Motor motor2(10);   // Constructor called with 10
// Both motors now initialized and ready to use

Initializer Lists (C++ Style)

Initializer lists are the preferred C++ way to initialize member variables:

class Sensor {
private:
    uint8_t pin;
    uint16_t reading;
    
public:
    // Initializer list: initialize members before constructor body runs
    Sensor(uint8_t pin_number) : pin(pin_number), reading(0) {
        pinMode(pin, INPUT);
    }
};

// Alternative (less efficient): assignment in constructor
class SensorOld {
private:
    uint8_t pin;
    
public:
    SensorOld(uint8_t pin_number) {
        pin = pin_number;  // Assign after creation
    }
};

// The initializer list version is better because members are 
// initialized at their declaration, not assigned after

Destructors: Cleanup on Deletion

class DynamicBuffer {
private:
    uint8_t *buffer;
    uint16_t size;
    
public:
    // Constructor: allocate memory
    DynamicBuffer(uint16_t buffer_size) : size(buffer_size) {
        buffer = new uint8_t[size];
    }
    
    // Destructor: cleanup—runs when object is deleted
    ~DynamicBuffer() {
        delete[] buffer;  // Free memory
        buffer = NULL;
    }
};

// Usage
{
    DynamicBuffer buf(512);
    // ... use buf
}  // Destructor called here automatically
Warning: Memory Leaks

If you allocate memory in a constructor but forget to free it in a destructor, you leak memory. On embedded systems with limited RAM, this is critical to manage:

// BAD: No destructor—memory leak!
class BadBuffer {
    uint8_t *data;
public:
    BadBuffer() { data = new uint8_t[1024]; }
    // Forgot destructor—memory never freed!
};

// GOOD: Destructor cleans up
class GoodBuffer {
    uint8_t *data;
public:
    GoodBuffer() { data = new uint8_t[1024]; }
    ~GoodBuffer() { delete[] data; }
};

Embedded Patterns with Structs and Classes

Pattern 1: Peripheral Driver Abstraction

Wrap hardware registers and functionality in a class for clean, portable code:

class SPIBus {
private:
    uint32_t base_address;  // Hardware register base
    uint32_t speed;
    
public:
    SPIBus(uint32_t address, uint32_t spi_speed) 
        : base_address(address), speed(spi_speed) {
        configure_hardware();
    }
    
    void transmit(uint8_t *data, uint16_t length) {
        for (int i = 0; i < length; i++) {
            *(volatile uint8_t*)(base_address + 0x04) = data[i];
            while (!(*(volatile uint32_t*)(base_address) & 0x80));
        }
    }
    
    uint8_t receive() {
        return *(volatile uint8_t*)(base_address + 0x04);
    }
    
private:
    void configure_hardware() {
        // Setup SPI controller
    }
};

// Clean driver interface:
SPIBus spi(0x40013000, 1000000);
uint8_t data[] = {0xAA, 0xBB, 0xCC};
spi.transmit(data, 3);

Pattern 2: Finite State Machine

Structs organize state and transition functions:

struct State {
    uint8_t id;
    void (*on_enter)();
    void (*on_update)();
    void (*on_exit)();
};

class TrafficLight {
private:
    uint8_t current_state;  // RED=0, YELLOW=1, GREEN=2
    uint32_t state_timer;
    State states[3];
    
public:
    TrafficLight() : current_state(0), state_timer(0) {
        states[0] = {0, red_enter, NULL, red_exit};
        states[1] = {1, yellow_enter, NULL, yellow_exit};
        states[2] = {2, green_enter, NULL, green_exit};
    }
    
    void update() {
        if (states[current_state].on_update) {
            states[current_state].on_update();
        }
    }
    
    void transition_to(uint8_t new_state) {
        if (states[current_state].on_exit) {
            states[current_state].on_exit();
        }
        current_state = new_state;
        if (states[current_state].on_enter) {
            states[current_state].on_enter();
        }
    }
};

// Usage
TrafficLight light;
light.transition_to(0);  // Go to RED
light.update();

Pattern 3: Sensor Data with Calibration

class PressureSensor {
private:
    uint16_t raw_reading;
    float calibration_scale;
    int16_t calibration_offset;
    uint8_t adc_pin;
    
public:
    PressureSensor(uint8_t pin) : adc_pin(pin), 
                                   calibration_scale(1.0), 
                                   calibration_offset(0) {
        pinMode(adc_pin, INPUT);
    }
    
    void calibrate(float zero_pressure_reading) {
        raw_reading = analogRead(adc_pin);
        calibration_offset = raw_reading;
    }
    
    float read_pressure_kpa() {
        raw_reading = analogRead(adc_pin);
        int16_t calibrated = raw_reading - calibration_offset;
        return calibrated * calibration_scale / 100.0;
    }
};

// Usage in main loop
PressureSensor sensor(A0);
// Calibrate on startup:
sensor.calibrate(0);
// Then read:
float pressure = sensor.read_pressure_kpa();
Best Practice: Keep Classes Lightweight on Microcontrollers

Avoid heavy C++ features (virtual functions, inheritance, STL containers) on tiny microcontrollers. Use simple structs or lightweight classes instead:

  • Prefer structs with simple data and functions
  • Avoid virtual functions (they add memory overhead)
  • Avoid inheritance hierarchies (stack is limited)
  • Use fixed-size arrays, not dynamic vectors

Key Takeaways

  • Structs and classes bundle related data and functions: Instead of scattered variables, organize them into cohesive units
  • Struct = all public by default; class = all private by default: Use structs for simple data grouping, classes for encapsulation
  • Constructors initialize objects automatically: Ensure all member variables are set up correctly on creation
  • Destructors clean up resources: Free memory and reset hardware state when an object is destroyed
  • Embedded patterns use structs/classes for drivers, FSMs, and abstractions: They make embedded code cleaner and more maintainable
  • Know your struct size: You need sizeof(struct) for EEPROM storage and memory calculations
  • Keep it simple on microcontrollers: Avoid heavy OOP features; focus on lightweight, efficient code

Project 1: LED Controller Class

Goal: Create an LED controller class that manages multiple LEDs with different pins and brightness levels.

Requirements:

  • Create an LED class with pin, brightness, and state (on/off)
  • Implement constructor to initialize GPIO pin
  • Add methods: turn_on(), turn_off(), set_brightness()
  • Implement pulse-width modulation (PWM) for brightness control
  • Create 3 LED objects and control them independently
class LED {
private:
    uint8_t pin;
    uint8_t brightness;  // 0-255
    bool state;
    
public:
    LED(uint8_t led_pin) : pin(led_pin), brightness(0), state(false) {
        pinMode(pin, OUTPUT);
        digitalWrite(pin, LOW);
    }
    
    void turn_on() {
        state = true;
        analogWrite(pin, brightness);
    }
    
    void turn_off() {
        state = false;
        digitalWrite(pin, LOW);
    }
    
    void set_brightness(uint8_t level) {
        brightness = level;
        if (state) {
            analogWrite(pin, brightness);
        }
    }
    
    uint8_t get_brightness() const {
        return brightness;
    }
};

// Usage
class LEDArray {
private:
    LED led1;
    LED led2;
    LED led3;
    
public:
    LEDArray() : led1(9), led2(10), led3(11) {}
    
    void demo() {
        // Turn on all LEDs
        led1.turn_on();
        led2.turn_on();
        led3.turn_on();
        
        // Fade: brightness ramp 0-255
        for (int b = 0; b <= 255; b += 5) {
            led1.set_brightness(b);
            delay(20);
        }
        
        // Turn off
        led1.turn_off();
        led2.turn_off();
        led3.turn_off();
    }
};

void setup() {
    LEDArray leds;
    leds.demo();
}

void loop() {}

Project 2: State Machine Motor Controller

Goal: Create a motor controller using a state machine pattern (IDLE → ACCELERATE → RUNNING → DECELERATE → IDLE).

Requirements:

  • Define an enum for states: IDLE, ACCELERATE, RUNNING, DECELERATE
  • Create a Motor class with state, speed, target_speed, acceleration
  • Implement state transition logic in update() method
  • Gradual speed ramp (don't jump instantly to full speed)
  • Demo: set target speed and observe smooth acceleration
enum MotorState {
    IDLE,
    ACCELERATE,
    RUNNING,
    DECELERATE
};

class MotorController {
private:
    uint8_t pin_pwm;
    uint8_t pin_direction;
    MotorState state;
    uint8_t current_speed;
    uint8_t target_speed;
    uint8_t acceleration;
    
public:
    MotorController(uint8_t pwm, uint8_t dir) 
        : pin_pwm(pwm), pin_direction(dir), 
          state(IDLE), current_speed(0), 
          target_speed(0), acceleration(2) {
        pinMode(pin_pwm, OUTPUT);
        pinMode(pin_direction, OUTPUT);
    }
    
    void set_target_speed(uint8_t speed) {
        target_speed = speed;
        if (target_speed > 0 && state == IDLE) {
            state = ACCELERATE;
        }
    }
    
    void update() {
        switch (state) {
            case IDLE:
                current_speed = 0;
                analogWrite(pin_pwm, 0);
                break;
                
            case ACCELERATE:
                if (current_speed < target_speed) {
                    current_speed += acceleration;
                    if (current_speed > target_speed) {
                        current_speed = target_speed;
                    }
                }
                if (current_speed >= target_speed) {
                    state = RUNNING;
                }
                analogWrite(pin_pwm, current_speed);
                break;
                
            case RUNNING:
                if (target_speed == 0) {
                    state = DECELERATE;
                }
                break;
                
            case DECELERATE:
                if (current_speed > 0) {
                    current_speed -= acceleration;
                }
                if (current_speed == 0) {
                    state = IDLE;
                }
                analogWrite(pin_pwm, current_speed);
                break;
        }
    }
    
    uint8_t get_speed() const { return current_speed; }
    MotorState get_state() const { return state; }
};

void setup() {
    Serial.begin(9600);
}

MotorController motor(9, 8);

void loop() {
    // Simulate speed commands
    static unsigned long last_time = 0;
    unsigned long now = millis();
    
    if (now - last_time > 5000) {  // Every 5 seconds
        static uint8_t phase = 0;
        
        if (phase == 0) {
            motor.set_target_speed(255);  // Full speed
            Serial.println("Accelerating...");
        } else if (phase == 1) {
            motor.set_target_speed(0);    // Stop
            Serial.println("Decelerating...");
        }
        
        phase = (phase + 1) % 2;
        last_time = now;
    }
    
    motor.update();
    
    // Print speed every 500ms
    static unsigned long last_print = 0;
    if (millis() - last_print > 500) {
        Serial.print("Speed: ");
        Serial.println(motor.get_speed());
        last_print = millis();
    }
}

Project 3: Configuration Storage with Structs

Goal: Create a device configuration struct and save/load it from EEPROM.

Requirements:

  • Define a DeviceConfig struct with device_id, baud_rate, calibration_value, debug_enabled
  • Create methods to save_to_eeprom() and load_from_eeprom()
  • Handle struct size using sizeof()
  • Demo: Save config, simulate power cycle (manual EEPROM reset), load and verify
  • Print loaded values to confirm persistence
// Arduino with EEPROM library
#include 

struct DeviceConfig {
    uint8_t device_id;
    uint16_t baud_rate;  // In units of 100 (so 96 = 9600)
    int16_t calibration_offset;
    uint8_t debug_enabled;
};

class ConfigManager {
private:
    DeviceConfig config;
    const uint16_t eeprom_address = 0;
    
public:
    ConfigManager() {
        // Set defaults
        config.device_id = 1;
        config.baud_rate = 96;  // 9600
        config.calibration_offset = 0;
        config.debug_enabled = 1;
    }
    
    void save_to_eeprom() {
        // Write struct to EEPROM
        uint8_t *data = (uint8_t*)&config;
        for (uint16_t i = 0; i < sizeof(DeviceConfig); i++) {
            EEPROM.write(eeprom_address + i, data[i]);
        }
        Serial.println("Config saved to EEPROM");
        print_config();
    }
    
    void load_from_eeprom() {
        // Read struct from EEPROM
        uint8_t *data = (uint8_t*)&config;
        for (uint16_t i = 0; i < sizeof(DeviceConfig); i++) {
            data[i] = EEPROM.read(eeprom_address + i);
        }
        Serial.println("Config loaded from EEPROM");
        print_config();
    }
    
    void print_config() {
        Serial.print("Device ID: ");
        Serial.println(config.device_id);
        Serial.print("Baud Rate: ");
        Serial.println(config.baud_rate * 100);
        Serial.print("Calibration Offset: ");
        Serial.println(config.calibration_offset);
        Serial.print("Debug Enabled: ");
        Serial.println(config.debug_enabled);
        Serial.print("Config Size: ");
        Serial.print(sizeof(DeviceConfig));
        Serial.println(" bytes");
    }
    
    void set_device_id(uint8_t id) { config.device_id = id; }
    void set_calibration(int16_t offset) { config.calibration_offset = offset; }
    uint8_t get_device_id() const { return config.device_id; }
};

ConfigManager config_mgr;

void setup() {
    Serial.begin(9600);
    delay(1000);
    
    Serial.println("=== Device Configuration Demo ===");
    Serial.println();
    
    // Set some values
    config_mgr.set_device_id(42);
    config_mgr.set_calibration(50);
    
    // Save to EEPROM
    config_mgr.save_to_eeprom();
    Serial.println();
    
    // Simulate power cycle by creating new manager
    ConfigManager config_mgr2;
    Serial.println("After simulated power cycle:");
    config_mgr2.load_from_eeprom();
}

void loop() {
}
