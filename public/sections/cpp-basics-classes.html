Classes & Structs in C++

What's a struct or class? They're like blueprints (or templates) that bundle related information and functions together. Think of it like a real-world example:

Car as a struct/class:

  • Data (properties): color, speed, fuel level, engine temperature
  • Functions (behaviors): accelerate(), brake(), openDoor(), startEngine()

Instead of creating separate variables:

uint8_t car_color;
uint16_t car_speed;
int16_t car_temp;
void car_accelerate() { } // So confusing! Nothing grouped

Use a struct to group them:

  • Data + functions = organized, clean, reusable
  • Create multiple cars: Car car1, car2, car3; Each one has its own car_color and car_speed

Embedded examples: A Motor object (speed, direction, enable_func), a Sensor object (readings, calibration, read_func), a HAL (Hardware Abstraction Layer) for your specific microcontroller.

Structs Basics

A struct groups multiple variables and functions into one type.

// Define a struct
struct Point {
    int16_t x;
    int16_t y;
};

Point p;
p.x = 10;
p.y = 20;

Point origin = {0, 0};

Point *ptr = &p;
ptr->x = 5;

Struct Functions (Methods)

Structs can contain functions that operate on their data. These are member functions, also called methods.

struct Point {
    int16_t x;
    int16_t y;
    
    uint32_t distance_from_origin() {
        return (x*x) + (y*y);
    }
};

Point p = {3, 4};
uint32_t dist_squared = p.distance_from_origin();

struct Motor {
    uint8_t pwm_pin;
    uint8_t direction_pin;
    uint8_t speed;
    
    void set_speed(uint8_t new_speed) {
        speed = new_speed;
        write_pwm(pwm_pin, new_speed);
    }
    
    void forward() {
        direction_pin = HIGH;
    }
    
    void backward() {
        direction_pin = LOW;
    }
    
    void stop() {
        speed = 0;
        write_pwm(pwm_pin, 0);
    }
};

Motor left_motor = {5, 6, 0};
left_motor.set_speed(200);
left_motor.forward();

Using this Pointer

Inside member functions, the this pointer refers to the current object instance. It's automatically available and lets you access the object's data explicitly.

struct CircularBuffer {
    uint8_t data[256];
    uint8_t write_index;
    uint8_t read_index;
    uint8_t count;
    
    void put(uint8_t value) {
        if (this->count < 256) {
            this->data[this->write_index] = value;
            this->write_index = (this->write_index + 1) % 256;
            this->count++;
        }
    }
    
    uint8_t get() {
        if (this->count > 0) {
            uint8_t value = this->data[this->read_index];
            this->read_index = (this->read_index + 1) % 256;
            this->count--;
            return value;
        }
        return 0;
    }
    
    uint8_t is_empty() {
        return this->count == 0;
    }
};

CircularBuffer buffer;
buffer.put(42);
uint8_t val = buffer.get();

Structs vs Classes

In C++, the main difference is visibility (public vs private).

// Struct - members are public by default
struct PublicStruct {
    uint8_t value;
};

PublicStruct ps;
ps.value = 5;

class PrivateClass {
private:
    uint8_t value;
    
public:
    void set_value(uint8_t v) {
        value = v;
    }
    
    uint8_t get_value() {
        return value;
    }
};

PrivateClass pc;
pc.set_value(5);

When to Use Each

// Use struct when all data is public
struct Point {
    int16_t x;
    int16_t y;
};

Point p = {10, 20};  // Simple, no validation needed

// Use class when you need validation/control
class ValidatedPoint {
private:
    int16_t x, y;
    
public:
    void set_x(int16_t new_x) {
        if (new_x >= -1000 && new_x <= 1000) {
            x = new_x;
        }
    }
    
    int16_t get_x() {
        return x;
    }
};

// In embedded, often use struct for simplicity

Constructors and Initialization

Constructors automatically initialize struct/class members.

struct Timer {
    uint32_t start_time;
    uint32_t duration;
    
    Timer(uint32_t duration_ms) {
        start_time = get_ms();
        duration = duration_ms;
    }
    
    bool is_expired() {
        return (get_ms() - start_time) >= duration;
    }
};

Timer button_timer(500);
while (!button_timer.is_expired()) {
    // processing
}

struct Range {
    int16_t min;
    int16_t max;
    
    Range() {
        min = 0;
        max = 100;
    }
    
    Range(int16_t minimum, int16_t maximum) {
        min = minimum;
        max = maximum;
    }
};

Range r1;
Range r2(-50, 50);

Device Driver Example

Practical patterns for hardware abstraction.

class UART {
private:
    uint32_t baud_rate;
    uint8_t tx_buffer[256];
    uint8_t rx_buffer[256];
    
public:
    UART(uint32_t baud) {
        baud_rate = baud;
        init_uart(baud);
    }
    
    void send_byte(uint8_t byte) {
        while (is_tx_busy());
        write_uart(byte);
    }
    
    void send_string(const char *str) {
        while (*str) {
            send_byte(*str++);
        }
    }
    
    uint8_t read_byte() {
        while (!is_rx_available());
        return read_uart();
    }
};

UART serial(9600);
serial.send_string("Hello World\n");

class GPIO {
private:
    uint8_t pin;
    
public:
    GPIO(uint8_t pin_number) {
        pin = pin_number;
        configure_pin_as_output(pin);
    }
    
    void set_high() {
        write_pin(pin, 1);
    }
    
    void set_low() {
        write_pin(pin, 0);
    }
    
    void toggle() {
        write_pin(pin, !read_pin(pin));
    }
};

GPIO led(13);
led.set_high();
led.toggle();
led.set_low();

State Machines with Structs

Common embedded pattern: state machine encapsulated in a struct.

// Device state machine
enum State {
    IDLE = 0,
    RUNNING = 1,
    ERROR = 2,
    SHUTDOWN = 3
};

struct Device {
    State current_state;
    uint16_t error_code;
    uint32_t uptime_ms;
    
    void process() {
        switch (current_state) {
            case IDLE:
                if (should_start()) {
                    current_state = RUNNING;
                }
                break;
                
            case RUNNING:
                if (check_error()) {
                    error_code = get_error();
                    current_state = ERROR;
                } else if (should_shutdown()) {
                    current_state = SHUTDOWN;
                }
                uptime_ms++;
                break;
                
            case ERROR:
                if (user_acknowledged_error()) {
                    current_state = IDLE;
                }
                break;
                
            case SHUTDOWN:
                shutdown_hardware();
                break;
        }
    }
    
    const char *state_str() {
        switch (current_state) {
            case IDLE: return "IDLE";
            case RUNNING: return "RUNNING";
            case ERROR: return "ERROR";
            case SHUTDOWN: return "SHUTDOWN";
            default: return "UNKNOWN";
        }
    }
};

Device device = {IDLE, 0, 0};

void main_loop() {
    while (1) {
        device.process();
        delay_ms(1);
    }
}

Static Members

Shared across all instances of a struct/class.

struct Logger {
    static uint16_t message_count;  // Shared by all instances
    uint8_t level;  // Per-instance
    
    static void print_stats() {
        uart_print("Total messages: ");
        uart_print_int(message_count);
    }
    
    void log(const char *msg) {
        message_count++;  // All instances share this
        uart_print("[");
        uart_print_int(level);
        uart_print("] ");
        uart_print(msg);
        uart_print("\n");
    }
};

uint16_t Logger::message_count = 0;  // Initialize static

Logger logger1, logger2;
logger1.log("Starting");
logger2.log("Running");
Logger::print_stats();  // Shows 2 messages total

Inheritance (Careful in Embedded!)

One class can inherit from another. Use sparingly in embedded—it adds code size.

// Base class
class Peripheral {
protected:
    uint32_t base_address;
    
public:
    Peripheral(uint32_t addr) {
        base_address = addr;
    }
    
    virtual void init() = 0;  // Pure virtual
};

// Derived class
class GPIO : public Peripheral {
public:
    GPIO(uint32_t addr) : Peripheral(addr) {}
    
    void init() {
        volatile uint32_t *reg = (volatile uint32_t *)base_address;
        *reg = 0x01;  // Enable GPIO
    }
};

GPIO port_a(0x40020000);
port_a.init();

// In embedded, often prefer composition over inheritance
struct System {
    UART uart;
    GPIO led;
    ADC adc;
    
    System() : uart(9600), led(13), adc(A0) {}
};

Memory Layout and sizeof

Understand how much memory structs use.

struct Compact {
    uint8_t a;
    uint8_t b;
    uint8_t c;
};
// sizeof(Compact) = 3 bytes

struct WithPadding {
    uint8_t a;
    uint16_t b;
    uint8_t c;
};
// sizeof(WithPadding) = 6 bytes (not 4!)

// #pragma pack prevents padding (but slower)
#pragma pack(1)
struct Packed {
    uint8_t a;
    uint16_t b;
    uint8_t c;
};
#pragma pack()
// sizeof(Packed) = 4 bytes (tight packing)

// Word-align for hardware efficiency
struct Aligned {
    uint32_t value1;  // 4 bytes
    uint32_t value2;  // 4 bytes
};
// sizeof(Aligned) = 8 bytes (efficient)

Common Mistakes

  • Forgetting semicolon: struct/class definition ends with ;
  • Private members: In classes, members are private by default
  • Memory overhead: Virtual functions and inheritance add code size
  • Struct alignment: Padding added for alignment; check sizeof()
  • Passing copies: Passing struct by value copies all data
// WRONG - Missing semicolon
struct Point {
    int16_t x, y;
}  // ERROR - need semicolon here

// RIGHT
struct Point {
    int16_t x, y;
};

// WRONG - Forgetting object pointer in method
void process(Data &data) {
    data.value = 5;  // If not reference parameter
}

// RIGHT - Use pointer or reference
void process(Data *data) {
    data->value = 5;
}

void process(Data &data) {
    data.value = 5;
}

// WRONG - Expensive struct copy
void process_sensor(SensorReading reading) {

// RIGHT - Use pointer or const reference
void process_sensor(const SensorReading *reading) {

Project 1: LED Controller Struct

Objective: Create a struct to represent an LED with control functions.

What you'll learn: How structs bundle data and functions that work together.

Step by step:

  1. Create: struct LED { uint8_t pin; uint8_t brightness; }
  2. Add a constructor: LED(uint8_t p, uint8_t b) to initialize
  3. Add methods: turnOn(), turnOff(), setBrightness(uint8_t b)
  4. Create two LED objects and control them independently
struct LED {
    uint8_t pin;
    uint8_t brightness;
        pin = p;
        brightness = b;
        pinMode(pin, OUTPUT);
    }
struct LED {
public:
    uint8_t pin;
        digitalWrite(pin, HIGH);
    }
    
    void turnOff() {
        digitalWrite(pin, LOW);
    }
    
    void setBrightness(uint8_t b) {
        brightness = b;
        analogWrite(pin, brightness);
    }
};

// Create LED objects
LED red_led(13, 255);
LED green_led(12, 128);

void setup() {
    Serial.begin(9600);
}

void loop() {
    red_led.turnOn();
    delay(500);
    
    green_led.setBrightness(50);
    delay(500);
    
    red_led.turnOff();
    delay(500);
}

Why this works: Each LED object bundles its pin and brightness data with the methods that control it. This is much cleaner than separate arrays and functions. You can create as many LEDs as needed, each fully independent.

Project 2: Simple Motor State Machine

Objective: Create a struct to represent a motor with multiple states (off, spinning slow, spinning fast).

What you'll learn: How structs naturally represent state machines with data and behavior together.

Step by step:

  1. Create: struct Motor with fields: pin, current_speed, max_speed
  2. Add methods: start(), stop(), setSpeed(uint8_t s)
  3. Track state (is it running?)
  4. Create a motor and demonstrate state changes
struct Motor {
    uint8_t pin;
    uint8_t current_speed;
    uint8_t max_speed;
    uint8_t is_running;
    
    Motor(uint8_t p, uint8_t max_s) {
        pin = p;
        max_speed = max_s;
        current_speed = 0;
        is_running = 0;
        pinMode(pin, OUTPUT);
    }
    
    void start() {
        is_running = 1;
        analogWrite(pin, current_speed);
    }
    
    void stop() {
        is_running = 0;
        current_speed = 0;
        analogWrite(pin, 0);
    }
    
    void setSpeed(uint8_t speed) {
        if (speed > max_speed) {
            speed = max_speed;
        }
        current_speed = speed;
        if (is_running) {
            analogWrite(pin, current_speed);
        }
    }
};

Motor main_motor(9, 200);

void loop() {
    main_motor.start();
    main_motor.setSpeed(100);   // Slow
    delay(2000);
    
    main_motor.setSpeed(200);   // Fast
    delay(2000);
    
    main_motor.stop();
    delay(1000);
}

Why this works: The Motor struct keeps state (is_running, current_speed) and provides methods that change state safely. setSpeed includes validation—it won't exceed max_speed. This prevents bugs from invalid states.

Project 3: Temperature Sensor Wrapper

Objective: Create a struct that reads a temperature sensor and provides averaged, filtered readings.

What you'll learn: How structs wrap complex hardware behavior in a clean interface.

Step by step:

  1. Create: struct TempSensor with pin, last_reading, min_temp, max_temp
  2. Add methods: read(), getAverage(), exceeded()
  3. Use a simple moving average technique (keep last 3 readings)
  4. Track min and max temperatures seen
struct TempSensor {
    uint8_t pin;
    uint8_t last_temp;
    uint8_t min_temp;
    uint8_t max_temp;
    uint8_t sample_count;
    
    TempSensor(uint8_t p) {
        pin = p;
        last_temp = 0;
        min_temp = 255;
        max_temp = 0;
        sample_count = 0;
        pinMode(pin, INPUT);
    }
    
    void read() {
        uint16_t raw = analogRead(pin);
        uint8_t temp = (raw / 1023) * 100;  // Scale to 0-100°C
        last_temp = temp;
        sample_count++;
        
        // Track min/max
        if (temp < min_temp) min_temp = temp;
        if (temp > max_temp) max_temp = temp;
    }
    
    uint8_t getLastReading() {
        return last_temp;
    }
    
    uint8_t getMin() {
        return min_temp;
    }
    
    uint8_t getMax() {
        return max_temp;
    }
    
    uint8_t getSampleCount() {
        return sample_count;
    }
};

TempSensor sensor(A0);

void loop() {
    sensor.read();
    
    Serial.print("Temp: ");
    Serial.print(sensor.getLastReading());
    Serial.print("°C  Min: ");
    Serial.print(sensor.getMin());
    Serial.print("  Max: ");
    Serial.print(sensor.getMax());
    Serial.print("  Samples: ");
    Serial.println(sensor.getSampleCount());
    
    delay(1000);
}

Why this works: The TempSensor struct encapsulates all sensor behavior. The outside code doesn't care how it reads or calculates—it just calls read() and asks for the data it needs. If you later add averaging logic, only the struct changes; the calling code stays the same.

Summary

  • Structs group related data and functions
  • Structs are public-by-default, classes are private-by-default
  • Use structs for simple data, classes for complex objects with validation
  • Constructors initialize members automatically
  • Perfect for hardware abstraction and device drivers
  • Static members are shared by all instances
  • Be aware of struct padding; check sizeof()
  • Use pointers/references to avoid expensive copies
  • State machines fit naturally in structs
